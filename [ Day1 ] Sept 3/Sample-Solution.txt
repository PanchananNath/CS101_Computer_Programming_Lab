Q5.

#include <stdio.h>

int main() {
    float circumference, radius, area;
    const float PI = 3.14;

    printf("Enter circumference of circle: ");
    scanf("%f", &circumference);

    radius = circumference / (2 * PI);
    area = PI * radius * radius;
    printf("Area = %.2f\n", area);
    return 0;
}
